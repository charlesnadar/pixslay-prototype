function fillMomB(name) { const inp = document.getElementById('momB-input'); const lbl = document.getElementById('momB-lbl'); if (inp) { inp.value = name; inp.style.border = '0.5px solid #C7D4FB'; if (lbl) lbl.style.cssText = 'position:absolute;left:14px;top:0;transform:translateY(-50%);font-size:11px;font-weight:600;color:#A8A49F;background:#fff;padding:0 4px;pointer-events:none;transition:all .15s;'; momBInputChange(); } } function momBInputChange() { const inp = document.getElementById('momB-input'); const btn = document.getElementById('momB-confirm'); if (!inp||!btn) return; if (inp.value.trim().length>0) btn.classList.remove('dis'); else btn.classList.add('dis'); } function confirmMomentB() { const inp = document.getElementById('momB-input'); const name = inp?.value.trim(); if (!name) return; state.identifiedFaces.push({index:currentMomentBIndex, name}); closeSheet('momB'); showToast(name+' identified!'); setTimeout(()=>{ const remaining = UNID_FACE_INDICES.filter(i => !state.identifiedFaces.find(x=>x.index===i-1)); if (remaining.length > 0) openMomentB(0); }, 1000); } function skipMomentB() { state.identifiedFaces.push({index:currentMomentBIndex, name:''}); closeSheet('momB'); setTimeout(()=>{ const remaining = UNID_FACE_INDICES.filter(i => !state.identifiedFaces.find(x=>x.index===i-1)); if (remaining.length > 0) openMomentB(0); }, 500); }
